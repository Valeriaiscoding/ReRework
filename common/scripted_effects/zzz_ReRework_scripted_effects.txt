ai_build_river_port = {
	every_state = {
		limit = { 
			NOT = { has_modifier = ai_river_port_cooldown }
			ia_rp_on_river = yes
			OR = {
				random_scope_building = {
					is_building_type = "building_ia_river_port"
					can_queue_building_levels = 1
				}
				NOT = { has_building = "building_ia_river_port" }
			}
		 }
		 add_modifier =  {
				name = ai_river_port_cooldown
				months = 4
		 }
		 if = {
			limit = { has_building = "building_ia_river_port" }
			random_scope_building = {
				limit = { is_building_type = "building_ia_river_port" }
				prev = {
					set_variable = river_port_fix_var
					change_variable = {
						name = river_port_fix_var
						add = {
							value = prev.level
							add = 1
						}
					}
					create_building = {
						building = "building_ia_river_port"
						level = var:river_port_fix_var
					}
					remove_variable = river_port_fix_var
				}
			}
		}
		else = {
			create_building = {
				building = "building_ia_river_port"
				level = 1
			}
		}
 }
  
}
ai_build_local_stuff = {
	every_state = {
		limit = { 
			NOT = { has_building = building_local_stuff_headquarters }
			has_building = building_barracks
			is_incorporated = yes
		 }
			random_scope_building = {
				limit = { is_building_type = "building_barracks" }
				prev = {
					set_variable = local_stuff_fix_var
					change_variable = {
						name = local_stuff_fix_var
						add = {
							value = prev.level
							subtract = 1
						}
					}
					create_building = {
						building = "building_barracks"
						level = var:local_stuff_fix_var
					}
					remove_variable = local_stuff_fix_var
				}
			}
			create_building = {
				building = "building_local_stuff_headquarters"
				level = 1
			}
 }
}
