je_baltic_unification = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	is_shown_when_inactive = {
		OR = {
			country_has_primary_culture = cu:estonian
			country_has_primary_culture = cu:latvian
			country_has_primary_culture = cu:lithuanian
		}
	}

	possible = {
		country_rank >= rank_value:minor_power
		has_technology_researched = pan-nationalism
		is_subject = no
	}

	complete = {
		calc_true_if = {
			amount > 2
			owns_entire_state_region = STATE_ESTONIA
			owns_entire_state_region = STATE_LATVIA
			owns_entire_state_region = STATE_LITHUANIA
		}
	}

	on_monthly_pulse = {
		events = {
			baltic_unification.2 # confederation
			baltic_unification.3 # Joudikis
		}
	}

	on_complete = {
		trigger_event = { id = baltic_unification.1 }
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_baltic_status
			}
		}
	}

	weight = 100
}
