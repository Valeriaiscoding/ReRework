

law_serfdom = {
	group = lawgroup_land_ownership

	icon = "gfx/interface/icons/law_icons/serfdom.dds"

	progressiveness = -50

	disallowing_laws = {
		law_interventionism
		law_laissez_faire
		law_command_economy
		law_smallholding
		law_land_redistribution
		law_land_collectivized
	}

	modifier = {
		state_aristocrats_investment_pool_efficiency_mult = -0.5
		state_infrastructure_from_population_max_mult = -0.2
		country_aristocrats_pol_str_mult = 1 #was none
		country_authority_add = 150 #was none
		country_tech_spread_mult = -0.1
		country_institution_schools_max_investment_add = -4
		state_peasants_migration_push_mult = -0.75
		tax_land_add = 0.1
	}

	possible_political_movements = {
		law_serfdom_banned
	}

	# AI should never try to bring back serfdom
	ai_will_do = {
		always = no
	}
}

law_serfdom_banned = {
	group = lawgroup_land_ownership

	icon = "gfx/interface/icons/law_icons/serfdom_banned.dds"

	progressiveness = 0

	modifier = {
    interest_group_ig_landowners_pol_str_mult = 0.15
    country_aristocrats_pol_str_mult = 0.2
    state_aristocrats_investment_pool_efficiency_mult  = 0.15
	}

	possible_political_movements = {
    law_smallholding
    law_land_collectivized
	}

	pop_support = {
		value = 0

		add = {
			desc = "POP_PEASANTS"
			if = {
				limit = {
					owner = {
						has_law = law_type:law_serfdom
					}
					is_pop_type = peasants
				}
				value = 1
				multiply = literacy_rate
			}
		}
		add = {
			desc = "POP_ARISTOCRATS"
			if = {
				limit = {
					is_pop_type = aristocrats
					owner = { 
						has_law = law_serfdom_banned
				}
				}
			}
			value = 0.25
			#multiply = literacy_rate
		}
	}
}

law_smallholding = {
	group = lawgroup_land_ownership

	icon = "gfx/interface/icons/law_icons/land_redistribution.dds"

	progressiveness = 50

	modifier = {
      state_migration_pull_mult = 0.3
      state_migration_pull_add = 3
	  interest_group_ig_landowners_pop_attraction_mult = -0.15
	}

  disallowing_laws = {
		law_traditionalism
	}

	possible_political_movements = {
    	law_serfdom_banned
		law_land_redistribution
    	law_land_collectivized
	}

	pop_support = {
		value = 0

		add = {
			desc = "POP_PEASANTS"
			if = {
				limit = {
					owner = {
						has_law = law_type:law_serfdom_banned
						NOT = { has_law = law_type:law_council_republic }
					}
					is_pop_type = peasants
				}
				value = 0.75
				multiply = literacy_rate
			}
		}
		add = {
			desc = "POP_ARISTOCRATS"
			if = {
				limit = {
					is_pop_type = aristocrats
					owner = { 
						has_law = law_land_redistribution
				}
				}
			}
			value = 0.25
			#multiply = literacy_rate
		}

		add = {
			desc = "POP_RURAL_EMPLOYMENT"
			if = {
				limit = { 
					NOT = { strata = rich }
					pop_employment_building_group = bg_agriculture
					owner = { NOT = { has_law = law_type:law_old_age_pension } }
				}
				
				# SoL 14 gives 0.1 pop support, 13 -> 0.2, 12 -> 0.3, ... 5 -> 1.0
				value = 15
				subtract = standard_of_living
				multiply = 0.1
				multiply = literacy_rate
				min = 0
				max = 1
			}				
		}
	}
}
