law_trinket_subsidies = {
	group = lawgroup_unemployment_subsidies
	
	icon = "gfx/interface/icons/law_icons/wealth_voting.dds"
	
	progressiveness = 50
	
	modifier = {
		state_welfare_payments_add = 0.1
		state_migration_pull_mult = 0.05
		interest_group_ig_trade_unions_pop_attraction_mult = 0.05
		interest_group_ig_trade_unions_pol_str_mult = 0.05
	}
	
	can_enact = { 
		has_technology_researched = labor_movement
	}
	
	pop_support = {
		value = 0
		
		add = {
			desc = "POP_POVERTY"			
			if = {
				limit = { 
					owner = { NOT = { has_law = law_type:law_trinket_subsidies } }
				}
				
				# SoL 19 gives 0.1 pop support, 18 -> 0.2, 17 -> 0.3, ... 5 -> 1.5
				value = 20
				subtract = standard_of_living
				multiply = 0.1
				multiply = literacy_rate
				min = 0
				max = 0.5
			}
		}
	}
}
