can_start_building_vineyard = {
	scope = country
	
    is_valid = {
		has_technology_researched = enclosure
		var:state_to_invest_in = {
			has_building = building_vineyard
			free_arable_land > 0
			owner = {
				has_technology_researched = enclosure
			}
		}
    }
}

can_start_building_ia_river_port = {
	scope = country
	
    is_valid = {
		has_technology_researched = navigation
		var:state_to_invest_in = {
			ia_rp_on_river = yes
			OR = {
				NOT = {
					has_building = building_ia_river_port
				}
				any_scope_building = {
					is_building_type = building_ia_river_port
					level < state.state_river_port_max_level
				}
			}
			owner = {
				has_technology_researched = navigation
			}
		}
    }
}
can_start_building_hydroelectric_power_plant = {
	scope = country
	
    is_valid = {
		has_technology_researched = electrical_generation
		
		var:state_to_invest_in = {
			on_river=yes
			OR = {
				NOT = {
					has_building = building_hydroelectric_power_plant
				}
				any_scope_building = {
					is_building_type = building_hydroelectric_power_plant
					level < country.max_hydro_power_level
					}
				}
			}
			owner = {
				has_technology_researched = electrical_generation
			}
		}
    }
